- [ ] **3.3** `components/deadline/SummarySection.tsx` — getSummary 호출 + 로딩/에러 처리 컨테이너
  ```typescript
  'use client';
  import { useState, useTransition } from 'react';
  import { getSummary } from '@/app/actions/deadline';
  import { SummaryCardGrid } from './SummaryCardGrid';
  import { Button } from '@/components/ui/button';
  import { Skeleton } from '@/components/ui/skeleton';
  import type { SummaryResult } from '@/lib/types/deadline';
  import * as Sentry from '@sentry/nextjs';

  export function SummarySection({ diagnosisId }: { diagnosisId: string }) {
    const [result, setResult] = useState<SummaryResult | null>(null);
    const [isPending, startTransition] = useTransition();
    const [error, setError] = useState<string | null>(null);

    function handleClick() {
      startTransition(async () => {
        try {
          setError(null);
          const data = process.env.NEXT_PUBLIC_USE_MOCK === 'true'
            ? (await import('@/lib/mocks/deadline')).MOCK_SUMMARY_RESULT
            : await getSummary(diagnosisId);
          setResult(data);
        } catch (e) {
          Sentry.captureException(e);
          setError('요약 생성 중 오류가 발생했습니다.');
        }
      });
    }

    return (
      <div className="space-y-4">
        {!result && (
          <Button onClick={handleClick} disabled={isPending} id="btn-30min-summary">
            {isPending ? '요약 생성 중...' : '30분 요약'}
          </Button>
        )}
        {isPending && (
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
            {[1, 2, 3].map(i => <Skeleton key={i} className="h-64 rounded-lg" />)}
          </div>
        )}
        {error && <p className="text-destructive" role="alert">{error}</p>}
        {result && <SummaryCardGrid cards={result.cards} />}
      </div>
    );
  }
  ```
