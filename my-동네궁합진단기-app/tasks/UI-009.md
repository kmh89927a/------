```
User-->Web: 이사 마감일 입력
alt 과거 날짜 입력
    Web-->Web: 클라이언트 검증 (≤ 100ms)
    Web-->User: "마감일은 오늘 이후여야 합니다" 인라인 에러
    Note: Server Action 호출 차단
else 미래 날짜 입력
    User-->Web: "데드라인 모드" 활성화 클릭
    Web-->SA: createDiagnosis(deadline_mode=true, deadline_date)
    SA-->SA: 계약 역산 타임라인 생성 (≥5단계)
    SA-->Web: 타임라인 응답 (≤2초)
    Web-->User: 계약 역산 타임라인 표시
end

  'use client';
  import { useState } from 'react';
  import { Calendar } from '@/components/ui/calendar';
  import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
  import { Button } from '@/components/ui/button';
  import { format } from 'date-fns';
  import { ko } from 'date-fns/locale';
  import { CalendarIcon } from 'lucide-react';
  import { cn } from '@/lib/utils';

  interface DeadlinePickerProps {
    value: Date | undefined;
    onChange: (date: Date | undefined) => void;
    disabled?: boolean;
  }

  const MIN_DEADLINE_DAYS = 7;

  export function DeadlinePicker({ value, onChange, disabled }: DeadlinePickerProps) {
    const [open, setOpen] = useState(false);
    const minDate = new Date(Date.now() + MIN_DEADLINE_DAYS * 24 * 60 * 60 * 1000);

    return (
      <Popover open={open} onOpenChange={setOpen}>
        <PopoverTrigger asChild>
          <Button
            variant="outline"
            className={cn(
              'w-full justify-start text-left font-normal',
              !value && 'text-muted-foreground'
            )}
            disabled={disabled}
            aria-label="이사 마감일 선택"
          >
            <CalendarIcon className="mr-2 h-4 w-4" />
            {value ? format(value, 'yyyy년 MM월 dd일', { locale: ko }) : '이사 마감일을 선택하세요'}
          </Button>
        </PopoverTrigger>
        <PopoverContent className="w-auto p-0" align="start">
          <Calendar
            mode="single"
            selected={value}
            onSelect={(date) => {
              onChange(date);
              setOpen(false);
            }}
            disabled={(date) => date < minDate}
            locale={ko}
            aria-label="이사 마감일 달력"
          />
        </PopoverContent>
      </Popover>
    );
  }
  ```

- [ ] **3.4** `components/deadline/TimelineCard.tsx` — 타임라인 카드 (UI-010 import용 export)
  ```typescript
  'use client';
  import type { TimelineDTO, TimelineStepDTO } from '@/lib/types/diagnosis';
  import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
  import { Badge } from '@/components/ui/badge';
  import { CheckCircle2, Circle } from 'lucide-react';

  interface TimelineCardProps {
    timeline: TimelineDTO;
  }

  export function TimelineCard({ timeline }: TimelineCardProps) {
    return (
      <Card className="w-full">
        <CardHeader>
          <CardTitle className="text-lg font-semibold">
            이사 일정 ({new Date(timeline.deadlineDate).toLocaleDateString('ko-KR')})
          </CardTitle>
        </CardHeader>
        <CardContent>
          <ol className="relative border-l border-gray-300 ml-3 space-y-4" aria-label="계약 역산 타임라인">
            {timeline.steps.map((step) => (
              <TimelineStep key={step.order} step={step} deadlineDate={timeline.deadlineDate} />
            ))}
          </ol>
        </CardContent>
      </Card>
    );
  }

  function TimelineStep({ step, deadlineDate }: { step: TimelineStepDTO; deadlineDate: string }) {
    const dueDate = new Date(step.dueDate);
    const deadline = new Date(deadlineDate);
    const daysUntil = Math.ceil((deadline.getTime() - dueDate.getTime()) / (1000 * 60 * 60 * 24));
    const dayLabel = daysUntil > 0 ? `D-${daysUntil}` : 'D-Day';

    return (
      <li className="ml-6 flex flex-col gap-1">
        <div className="absolute -left-3 flex items-center justify-center w-6 h-6 rounded-full bg-white border-2 border-primary">
          {step.completed ? (
            <CheckCircle2 className="w-4 h-4 text-green-600" aria-label="완료됨" />
          ) : (
            <Circle className="w-4 h-4 text-gray-400" aria-label="미완료" />
          )}
        </div>
        <div className="flex items-center gap-2">
          <Badge variant={daysUntil <= 3 ? 'destructive' : 'secondary'}>
            {dayLabel}
          </Badge>
          <span className="font-medium text-sm">{step.title}</span>
        </div>
        <p className="text-xs text-muted-foreground">{step.description}</p>
        <time className="text-xs text-muted-foreground" dateTime={step.dueDate}>
          {dueDate.toLocaleDateString('ko-KR')}
        </time>
      </li>
    );
  }
  ```

- [ ] **3.6** `components/deadline/DeadlineInputForm.tsx` — react-hook-form + Zod 폼 구현
  ```typescript
  'use client';
  import { useState } from 'react';
  import { useForm } from 'react-hook-form';
  import { zodResolver } from '@hookform/resolvers/zod';
  import { z } from 'zod';
  import { activateDeadlineMode } from '@/app/actions/deadline';
  import { DeadlinePicker } from './DeadlinePicker';
  import { TimelineCard } from './TimelineCard';
  import { Button } from '@/components/ui/button';
  import { useToast } from '@/components/ui/use-toast';
  import type { TimelineDTO } from '@/lib/types/diagnosis';
  import { reportErrorToSentry } from '@/lib/helpers/app-error';

  const deadlineFormSchema = z.object({
    deadlineDate: z.date({ required_error: '이사 마감일을 선택해주세요' }),
  });

  type DeadlineFormValues = z.infer<typeof deadlineFormSchema>;

  interface DeadlineInputFormProps {
    diagnosisId?: string;
  }

  export function DeadlineInputForm({ diagnosisId }: DeadlineInputFormProps) {
    const [timeline, setTimeline] = useState<TimelineDTO | null>(null);
    const [isLoading, setIsLoading] = useState(false);
    const { toast } = useToast();

    const form = useForm<DeadlineFormValues>({
      resolver: zodResolver(deadlineFormSchema),
    });

    async function handleSubmit(values: DeadlineFormValues) {
      if (!diagnosisId) return;
      setIsLoading(true);

      try {
        const result = await activateDeadlineMode({
          diagnosisId,
          deadlineDate: values.deadlineDate.toISOString(),
        });
        setTimeline(result);
        toast({ title: '타임라인이 생성되었습니다' });
      } catch (error: any) {
        if (error?.code === 'DIAG_DEADLINE_TOO_SOON') {
          toast({
            variant: 'destructive',
            title: '이사일은 7일 이후로 선택해주세요',
            description: '최소 7일의 준비 기간이 필요합니다.',
          });
        } else {
          toast({
            variant: 'destructive',
            title: '오류가 발생했습니다',
            description: '잠시 후 다시 시도해주세요.',
          });
          reportErrorToSentry(error instanceof Error ? error : new Error('Deadline form error'));
        }
      } finally {
        setIsLoading(false);
      }
    }

    return (
      <div className="space-y-6">
        <form onSubmit={form.handleSubmit(handleSubmit)} className="space-y-4">
          <div>
            <label className="block text-sm font-medium mb-2" htmlFor="deadline-picker">
              이사 마감일
            </label>
            <DeadlinePicker
              value={form.watch('deadlineDate')}
              onChange={(date) => form.setValue('deadlineDate', date as Date)}
              disabled={isLoading}
            />
            {form.formState.errors.deadlineDate && (
              <p className="text-destructive text-sm mt-1" role="alert">
                {form.formState.errors.deadlineDate.message}
              </p>
            )}
          </div>
          <Button
            type="submit"
            className="w-full"
            disabled={isLoading || !form.watch('deadlineDate')}
            aria-label="데드라인 모드 활성화"
          >
            {isLoading ? '타임라인 생성 중...' : '데드라인 모드 활성화'}
          </Button>
        </form>

        {timeline && <TimelineCard timeline={timeline} />}
      </div>
    );
  }
  ```

## 9. 🚧 Open Questions / Risks

1. **Calendar locale 한국어 설정:** shadcn Calendar은 `date-fns` 기반. `ko` locale import 필요 (`import { ko } from 'date-fns/locale'`). shadcn 버전에 따라 locale prop 지원 방식 상이 — 설치 시 확인.
2. **D+7 최소값 근거:** CMD-DL-001에서 D+7 최소 기간 설정. SRS REQ-FUNC-020은 "과거 날짜 차단"만 명시. D+7은 실무적 최소 계약 기간 기준으로 CMD-DL-001과 일치.
3. **diagnossId 전달 방식:** DeadlineInputForm에 `diagnosisId` prop 필요. 진단 완료 후 페이지 라우팅으로 전달 (URL params 또는 context). 라우팅 패턴은 UI-010과 함께 확정.
4. **TimelineCard 스타일 커스터마이징:** D-3 이내 단계에 `destructive` 배지 적용. 색상/크기는 디자인 시스템 확정 시 조정.
